-- User Table
CREATE TABLE User (
    UserID INT PRIMARY KEY AUTO_INCREMENT,
    Email VARCHAR(255) UNIQUE NOT NULL,
    Password VARCHAR(255) NOT NULL
    -- Other attributes can be added here if needed
);

-- Admin Table
CREATE TABLE Admin (
    AdminID INT PRIMARY KEY AUTO_INCREMENT,
    UserID INT,
    FOREIGN KEY (UserID) REFERENCES User(UserID)
    -- Additional admin-specific attributes can go here
);

-- Company Table
CREATE TABLE Company (
    CompanyID INT PRIMARY KEY AUTO_INCREMENT,
    UserID INT,
    Name VARCHAR(255) NOT NULL,
    Address TEXT,
    Description TEXT,
    Industry VARCHAR(255),
    CompanyLogo LONGBLOB,
    FOREIGN KEY (UserID) REFERENCES User(UserID)
    -- You might want to store the path to the logo instead of the binary data
);

-- Student Table
CREATE TABLE Student (
    StudentID INT PRIMARY KEY AUTO_INCREMENT,
    UserID INT,
    FirstName VARCHAR(255),
    LastName VARCHAR(255),
    Faculty VARCHAR(255),
    Course VARCHAR(255),
    ProfilePhoto LONGBLOB,
    CV LONGBLOB,
    FOREIGN KEY (UserID) REFERENCES User(UserID)
    -- As with the logo, consider storing file paths instead of binary data
);
